{# Toast component (Alpine-powered) for Jinja2. Usage: {% from "components/toast.html" import toast, toast_oob %} {# render a toast inline (wherever you place it) #} {{ toast("Saved!", type="success") }} {# append globally via HTMX out-of-band swap (recommended) #} {{ toast_oob("Saved!", type="success") }} IMPORTANT: - Put the container ONCE in your base layout, e.g. in admin/base.html:
- Add global CSS once: [x-cloak] { display:none !important; } #} {% macro toast(message, type="info", duration=3000) %}
{% if type == 'success' %} {% elif type == 'error' %} {% elif type == 'warning' %} {% else %} {% endif %}
{{ message }}
{% endmacro %} {# Helper macro to append the toast into #toast-container via HTMX out-of-band swap. Use this in HTMX responses so you don't need to target the container manually. #} {% macro toast_oob(message, type="info", duration=3000) %}
{{ toast(message, type=type, duration=duration) }}
{% endmacro %}
